#include <stdio.h>

int main() {
    int n, i;
    unsigned long long fatorial = 1;

    printf("Digite um número inteiro positivo: ");
    scanf("%d", &n);

    if (n < 0) {
        printf("Não existe fatorial de número negativo.\n");
    } else {
        printf("%d! = ", n);
        
        for (i = n; i >= 1; i--) {
            fatorial *= i;
            
            // Apenas para formatar a saída igual ao seu exemplo
            if (i > 1) printf("%d.", i);
            else printf("%d", i);
        }

        printf(" = %llu\n", fatorial);
    }

    return 0;
}
